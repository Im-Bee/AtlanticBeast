FILE(GLOB_RECURSE CORE_SOURCE 
    "${CMAKE_CURRENT_SOURCE_DIR}/Private/*.*"
)
FILE(GLOB_RECURSE CORE_HEADRES 
    "${CMAKE_CURRENT_SOURCE_DIR}/Public/*.h"
    "${CMAKE_CURRENT_SOURCE_DIR}/Public/*.hpp"
)

ADD_LIBRARY(CoreLib 
    "${CORE_SOURCE}"
    "${CORE_HEADRES}"
)

FIND_PACKAGE(Vulkan REQUIRED)

TARGET_INCLUDE_DIRECTORIES(CoreLib PUBLIC 
    "${Vulkan_INCLUDE_DIRS}"
    "${CMAKE_CURRENT_SOURCE_DIR}/Public/"
)

TARGET_LINK_LIBRARIES(CoreLib PUBLIC 
    ${Vulkan_LIBRARIES}
)

TARGET_SOURCES(CoreLib PUBLIC
    "${CORE_SOURCE}"
    "${CORE_HEADRES}"
)
