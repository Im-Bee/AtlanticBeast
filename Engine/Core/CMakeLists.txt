FILE(GLOB_RECURSE CORE_SOURCE 
    "${CMAKE_CURRENT_SOURCE_DIR}/Private/*.*"
)
FILE(GLOB_RECURSE CORE_HEADRES 
    "${CMAKE_CURRENT_SOURCE_DIR}/Public/*.h"
    "${CMAKE_CURRENT_SOURCE_DIR}/Public/*.hpp"
)

ADD_LIBRARY(CoreLib 
    "${CORE_SOURCE}"
    "${CORE_HEADRES}"
)

IF (UNIX)
    FIND_PACKAGE(X11 REQUIRED)

    TARGET_INCLUDE_DIRECTORIES(CoreLib PRIVATE ${X11_INCLUDE_DIR})
    TARGET_LINK_LIBRARIES(CoreLib PRIVATE ${X11_LIBRARIES})
endif()

TARGET_INCLUDE_DIRECTORIES(CoreLib PUBLIC 
    "${CMAKE_CURRENT_SOURCE_DIR}/Public/"
)

TARGET_SOURCES(CoreLib PUBLIC
    "${CORE_SOURCE}"
    "${CORE_HEADRES}"
)
